And define:

def has_valid_channels_for_genre(genre):
    epg_json = os.path.join(CACHE_DIR, "epg.json")
    if not xbmcvfs.exists(epg_json):
        return False
    with open(epg_json, "r", encoding="utf-8") as f:
        data = json.load(f)
    return any(ch.get("genre") == genre for ch in data.get("channels", []))


That way, the next startup automatically detects and clears an empty genre filter.

    xbmcgui.Dialog().ok("Empty EPG", "No channels found for this filter.")
